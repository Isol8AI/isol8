# E2E Gate Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the manual prod deployment approval gate with an automated Playwright e2e suite that runs the full user journey (auth → subscribe → provision → chat → cleanup) against `https://dev.isol8.co` on every merge to `main`.

**Architecture:** A `GitHubActionStep("E2EGate")` in `apps/infra/lib/app.ts` replaces `ApproveProduction`. The step runs `npx playwright test --project=chromium` against the live dev site. Tests use Clerk Testing Tokens for auth, Stripe Node SDK for subscription management (bypassing browser UI), and the backend debug API for container provisioning.

**Tech Stack:** Playwright, `@clerk/testing/playwright`, Stripe Node SDK (`stripe ^17`), Vitest (unit tests for helpers), `cdk-pipelines-github` (CDK pipeline), pnpm workspaces.

**Spec:** `docs/superpowers/specs/2026-04-01-e2e-gate-design.md`

```typescript
const POLL_INTERVAL_MS = 3000;

/**
 * DELETE /api/v1/debug/provision — safe to call when no container exists.
 * Treats 404 (no container) and 503 (ECS error on nonexistent service) as "not running".
 */
export async function deprovisionIfExists(apiUrl: string, authToken: string): Promise<void> {
  const res = await fetch(`${apiUrl}/debug/provision`, {
    method: 'DELETE',
    headers: { Authorization: `Bearer ${authToken}` },
  });
  if (!res.ok && res.status !== 404 && res.status !== 503) {
    throw new Error(`Unexpected deprovision response: ${res.status}`);
  }
}

/**
 * Poll GET /api/v1/container/status until status === "running".
 * Throws on status === "error" or timeout.
 */
export async function waitForRunning(
  apiUrl: string,
  authToken: string,
  timeoutMs: number,
): Promise<void> {
  const deadline = Date.now() + timeoutMs;
  while (Date.now() < deadline) {
    const res = await fetch(`${apiUrl}/container/status`, {
      headers: { Authorization: `Bearer ${authToken}` },
    });
    if (res.ok) {
      const data = await res.json();
      if (data.status === 'running' || data.substatus === 'gateway_healthy') return;
      if (data.status === 'error') {
        throw new Error(`Container entered error state: ${data.last_error}`);
      }
    }
    await new Promise((r) => setTimeout(r, POLL_INTERVAL_MS));
  }
  throw new Error(`waitForRunning: timeout after ${timeoutMs}ms`);
}
```

```typescript
import Stripe from 'stripe';

const POLL_INTERVAL_MS = 5000;

function getStripe(secretKey: string): Stripe {
  return new Stripe(secretKey, { apiVersion: '2025-01-27.acacia' });
}

/**
 * Cancel any active/trialing/incomplete subscriptions for the given email.
 * Handles multiple Stripe customers with the same email.
 * Safe to call when no subscription exists.
 *
 * Note: status filter is omitted from subscriptions.list — 'all' is not valid in
 * API version 2025-01-27.acacia. Status is checked client-side instead.
 */
export async function cancelSubscriptionIfExists(email: string, secretKey: string): Promise<void> {
  const stripe = getStripe(secretKey);
  const customers = await stripe.customers.list({ email, limit: 10 });
  for (const customer of customers.data) {
    const subscriptions = await stripe.subscriptions.list({
      customer: customer.id,
      limit: 10,
      // No status filter — check client-side below
    });
    for (const sub of subscriptions.data) {
      if (['active', 'trialing', 'incomplete'].includes(sub.status)) {
        await stripe.subscriptions.cancel(sub.id);
      }
    }
  }
}

/**
 * Create a Stripe subscription for the given email using pm_card_visa.
 * Sets metadata.plan_tier = "starter" so the backend webhook maps it correctly.
 * Looks up existing Stripe customer by email; creates one if not found.
 */
export async function createSubscription(
  email: string,
  priceId: string,
  secretKey: string,
): Promise<void> {
  const stripe = getStripe(secretKey);

  const customers = await stripe.customers.list({ email, limit: 1 });
  let customerId: string;

  if (customers.data.length > 0) {
    customerId = customers.data[0].id;
  } else {
    const customer = await stripe.customers.create({ email });
    customerId = customer.id;
  }

  await stripe.paymentMethods.attach('pm_card_visa', { customer: customerId });
  await stripe.customers.update(customerId, {
    invoice_settings: { default_payment_method: 'pm_card_visa' },
  });

  await stripe.subscriptions.create({
    customer: customerId,
    items: [{ price: priceId }],
    default_payment_method: 'pm_card_visa',
    metadata: { plan_tier: 'starter' }, // read by billing.py webhook handler
  });
}

/**
 * Poll GET /api/v1/billing/account until is_subscribed === true.
 * Bridges the async gap between Stripe subscription creation and the backend
 * processing the customer.subscription.created webhook.
 */
export async function waitForSubscriptionActive(
  apiUrl: string,
  authToken: string,
  timeoutMs: number,
): Promise<void> {
  const deadline = Date.now() + timeoutMs;
  while (Date.now() < deadline) {
    const res = await fetch(`${apiUrl}/billing/account`, {
      headers: { Authorization: `Bearer ${authToken}` },
    });
    if (res.ok) {
      const data = await res.json();
      if (data.is_subscribed === true) return;
    }
    await new Promise((r) => setTimeout(r, POLL_INTERVAL_MS));
  }
  throw new Error(`waitForSubscriptionActive: timeout after ${timeoutMs}ms`);
}
```

## Task 6: Create Journey Spec

**Files:**
- Create: `apps/frontend/tests/e2e/journey.spec.ts`

Design decisions:
- `test.use({ retries: 0 })` is **load-bearing** — the global config has `retries: 2` in CI. Removing `test.use({ retries: 0 })` would cause Playwright to retry the journey tests, re-running destructive operations (provision, subscribe) on already-provisioned state. Do not remove it.
- Auth token is stored in a module-level variable so `afterAll` can reuse it without re-signing in.
- `clerk.signIn()` requires `page.goto()` to be called first — Clerk's JS must be loaded before `signIn()` polls for `window.Clerk`. Always navigate to a non-protected page (e.g., `/`) before calling `clerk.signIn()`.
- The shared page is created in `beforeAll`, signed in once, and reused across all serial steps.

- [ ] **Step 1: Create the journey spec**

Create `apps/frontend/tests/e2e/journey.spec.ts`:

```typescript
import { test, expect, Browser, Page } from '@playwright/test';
import { clerk } from '@clerk/testing/playwright';
import {
  cancelSubscriptionIfExists,
  createSubscription,
  waitForSubscriptionActive,
} from './helpers/stripe';
import { deprovisionIfExists, waitForRunning } from './helpers/provision';

// ─── Constants ────────────────────────────────────────────────────────────────
const DEV_STARTER_PRICE_ID = 'price_1TF5MDI54BysGS3rlT80MMI8'; // from service-stack.ts
const API_URL = process.env.NEXT_PUBLIC_API_URL!;
const TEST_EMAIL = process.env.E2E_CLERK_USER_USERNAME!;
const TEST_PASSWORD = process.env.E2E_CLERK_USER_PASSWORD!;
const STRIPE_KEY = process.env.STRIPE_SECRET_KEY!;

// ─── Test config ──────────────────────────────────────────────────────────────
test.describe.configure({ mode: 'serial' });
// IMPORTANT: retries: 0 is load-bearing. The global config has retries: 2 in CI.
// Retrying would re-run destructive ops (subscribe, provision) on already-modified state.
test.use({ retries: 0 });

// Module-level state shared across serial steps
let sharedPage: Page;
let authToken: string;

// ─── Lifecycle ────────────────────────────────────────────────────────────────

test.beforeAll(async ({ browser }: { browser: Browser }) => {
  const context = await browser.newContext();
  sharedPage = await context.newPage();

  // page.goto() must be called before clerk.signIn() — Clerk JS must be loaded first
  await sharedPage.goto('/');
  await clerk.signIn({
    page: sharedPage,
    signInParams: {
      strategy: 'password',
      identifier: TEST_EMAIL,
      password: TEST_PASSWORD,
    },
  });

  // Store token for use in API calls throughout the journey
  authToken = await sharedPage.evaluate(
    async () => await (window as any).Clerk?.session?.getToken(),
  );
});

test.afterAll(async () => {
  // Best-effort cleanup — runs immediately after failure since retries: 0
  // Uses stored authToken (no re-sign-in needed).
  // If authToken is falsy, beforeAll failed at sign-in — nothing was provisioned or subscribed.
  // Check Clerk credentials, not Stripe/container state, if you see authToken undefined here.
  try {
    if (authToken) await deprovisionIfExists(API_URL, authToken);
  } catch {
    // Swallow cleanup errors — don't fail the suite on cleanup issues
  }
  try {
    await cancelSubscriptionIfExists(TEST_EMAIL, STRIPE_KEY);
  } catch {
    // Swallow
  }
  await sharedPage?.context().close();
});

// ─── Journey steps ────────────────────────────────────────────────────────────

test('step 1: idempotent cleanup', async () => {
  expect(authToken, 'Auth token must be available after sign-in').toBeTruthy();
  await deprovisionIfExists(API_URL, authToken);
  await cancelSubscriptionIfExists(TEST_EMAIL, STRIPE_KEY);
}, { timeout: 2 * 60 * 1000 });

test('step 2: auth — chat page is accessible', async () => {
  await sharedPage.goto('/chat');
  await expect(sharedPage).toHaveURL(/\/chat/);
  // Verify not redirected to sign-in
  await expect(sharedPage.locator('body')).not.toContainText('sign in', { ignoreCase: true });
}, { timeout: 30 * 1000 });

test('step 3: subscribe via stripe api and verify active in app', async () => {
  await createSubscription(TEST_EMAIL, DEV_STARTER_PRICE_ID, STRIPE_KEY);
  // Poll billing API until webhook is processed and plan_tier updated
  await waitForSubscriptionActive(API_URL, authToken, 60 * 1000);
}, { timeout: 90 * 1000 });

test('step 4: provision container and wait for running', async () => {
  // Use page.evaluate so the request carries the browser's auth cookies
  const result = await sharedPage.evaluate(
    async ({ apiUrl, token }: { apiUrl: string; token: string }) => {
      const res = await fetch(`${apiUrl}/debug/provision`, {
        method: 'POST',
        headers: { Authorization: `Bearer ${token}` },
      });
      return { ok: res.ok, status: res.status };
    },
    { apiUrl: API_URL, token: authToken },
  );
  expect(result.ok, `Provision failed with HTTP ${result.status}`).toBe(true);

  await waitForRunning(API_URL, authToken, 5 * 60 * 1000);
}, { timeout: 6 * 60 * 1000 }); // 5 min provision + 1 min buffer

test('step 5: chat — send a message and get a response', async () => {
  await sharedPage.goto('/chat');

  // Wait for the chat input (container connected)
  const chatInput = sharedPage.getByPlaceholderText('Ask anything');
  await expect(chatInput).toBeVisible({ timeout: 30 * 1000 });

  // Send a simple, deterministic message
  await chatInput.fill('Reply with exactly the word PONG and nothing else.');
  await sharedPage.getByTestId('send-button').click();

  // Verify streaming starts (stop button appears) then completes (stop button disappears)
  await expect(sharedPage.getByTestId('stop-button')).toBeVisible({ timeout: 10 * 1000 });
```
