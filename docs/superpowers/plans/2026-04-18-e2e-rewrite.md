# E2E Test Rewrite Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the shared-state, admin-API-heavy E2E gate with per-run isolated UI-driven tests (Personal + Org flows), each creating a fresh Clerk user, exercising real Stripe Checkout, and tearing down with hard-fail verification.

**Architecture:** Backend gains 3 `/debug/*` endpoints + a correlation-ID middleware. Frontend gains a Playwright fixture system (`fixtures/`, `drivers/`, `assertions/`) that admin-creates a Clerk user per worker, drives the real UI for sign-in / onboarding / Stripe Checkout / chat, then runs a hard-fail teardown verification. Two parallel describe-block flows (`personal.spec.ts`, `org.spec.ts`) replace the current `journey.spec.ts` + `chat.smoke.spec.ts`. CI wiring drops PR-time E2E and updates the post-deploy gate.

**Tech Stack:** FastAPI + Pydantic + boto3 (backend), Playwright + TypeScript (frontend tests), Vitest (unit tests for helpers), Clerk Backend API, Stripe Node SDK (test-mode), AWS DynamoDB + ECS + EFS, GitHub Actions (CI).

```python
# apps/backend/core/observability/e2e_correlation.py
"""Bind X-E2E-Run-Id header to per-request log context.

When the e2e harness issues a request with X-E2E-Run-Id, every log line
emitted while handling that request includes the same `e2e_run_id` field.
This lets ops grep CloudWatch by a single ID and see the full trace of
a failed e2e run.
"""

import contextvars
import logging
from typing import Awaitable, Callable

from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request
from starlette.responses import Response

# Per-request context. Reset to None at request boundaries so requests
# without the header don't inherit the previous request's run_id.
_e2e_run_id_var: contextvars.ContextVar[str | None] = contextvars.ContextVar(
    "e2e_run_id", default=None
)


class E2ERunIdLogFilter(logging.Filter):
    """Inject the current request's e2e_run_id (if any) into every log record."""

    def filter(self, record: logging.LogRecord) -> bool:
        run_id = _e2e_run_id_var.get()
        if run_id is not None:
            record.e2e_run_id = run_id
        return True


class E2ECorrelationMiddleware(BaseHTTPMiddleware):
    """FastAPI middleware: read X-E2E-Run-Id header, bind to log context.

    The contextvar is the binding mechanism so async work spawned inside
    the request handler inherits the value automatically.
    """

    def __init__(self, app):
        super().__init__(app)
        # Install the log filter once on the root logger. Idempotent —
        # safe to call multiple times during testing because filters are
        # de-duped by identity, not class.
        root = logging.getLogger()
        if not any(isinstance(f, E2ERunIdLogFilter) for f in root.filters):
            root.addFilter(E2ERunIdLogFilter())

    async def dispatch(
        self, request: Request, call_next: Callable[[Request], Awaitable[Response]]
    ) -> Response:
        run_id = request.headers.get("X-E2E-Run-Id")
        token = _e2e_run_id_var.set(run_id) if run_id else None
        try:
            return await call_next(request)
        finally:
            if token is not None:
                _e2e_run_id_var.reset(token)
```

```python
async def delete_all_for_user(self, user_id: str) -> int:
    """Delete all WS connection rows for a user (the user-id-index GSI lookup)."""
    response = await asyncio.to_thread(
        self._client.query,
        TableName=self.table_name,
        IndexName="user-id-index",
        KeyConditionExpression="user_id = :u",
        ExpressionAttributeValues={":u": {"S": user_id}},
    )
    items = response.get("Items", [])
    for item in items:
        await asyncio.to_thread(
            self._client.delete_item,
            TableName=self.table_name,
            Key={"connection_id": item["connection_id"]},
        )
    return len(items)
```

```python
@router.delete(

    description="Tears down ECS service, EFS access point, EFS folder, all "
                "per-user task-def revisions, and rows from all 8 per-user "
                "DDB tables. Used by the e2e harness at the end of each run. "

async def delete_user_data(auth: AuthContext = Depends(get_current_user)):
    if settings.ENVIRONMENT == "prod":
        raise HTTPException(status_code=403, detail="Disabled in production")

    owner_id = resolve_owner_id(auth)
    deleted = {"ecs": False, "efs": False, "ddb": []}

    # Container teardown — get the row before deleting so we have access_point_id + task_def_arn
    container = await container_repo.get_by_owner_id(owner_id)
    ecs_mgr = get_ecs_manager()
    if container:
        try:
            await ecs_mgr.delete_user_service(owner_id)
            deleted["ecs"] = True
        except Exception as e:
            logger.warning("ECS teardown failed for %s: %s", owner_id, e)
        # Deregister all task-def revisions (best-effort)
        if container.get("task_definition_arn"):
            try:
                ecs_mgr._deregister_task_definition(container["task_definition_arn"])
            except Exception:
                pass

    # EFS folder rm -rf (best-effort)
    try:
        get_workspace().delete_user_dir(owner_id)
        deleted["efs"] = True
    except Exception as e:
        logger.warning("EFS rm -rf failed for %s: %s", owner_id, e)

    # DDB cleanup, all 8 tables
    await container_repo.delete(owner_id)
    deleted["ddb"].append("containers")

    await billing_repo.delete(owner_id)
    deleted["ddb"].append("billing-accounts")

    await api_key_repo.delete_all_for_owner(owner_id)
    deleted["ddb"].append("api-keys")

    await usage_repo.delete_all_for_owner(owner_id)
    deleted["ddb"].append("usage-counters")

    await update_repo.delete_all_for_owner(owner_id)
    deleted["ddb"].append("pending-updates")

    await channel_link_repo.delete_all_for_owner(owner_id)
    deleted["ddb"].append("channel-links")

    await connection_service.delete_all_for_user(owner_id)
    deleted["ddb"].append("ws-connections")

    await user_repo.delete(owner_id)
    deleted["ddb"].append("users")

    return {"deleted": deleted}
```

```typescript
// apps/frontend/tests/e2e/fixtures/api.ts
import type { Page } from '@playwright/test';

/**
 * Page-aware authenticated fetch. Reads a fresh Clerk JWT from the page on
 * every call (Clerk tokens expire after 60s). Adds X-E2E-Run-Id for backend
 * log correlation.
 */
export class AuthedFetch {
  constructor(
    private page: Page,
    private apiUrl: string,
    private runId: string,
  ) {}

  private async token(): Promise<string> {
    return this.page.evaluate(async () => {
      const w = window as Window & {
        Clerk?: { session?: { getToken: () => Promise<string> } };
      };
      return (await w.Clerk?.session?.getToken()) ?? '';
    });
  }

  private async send(method: string, path: string, body?: unknown): Promise<Response> {
    const token = await this.token();
    return fetch(`${this.apiUrl}${path}`, {
      method,
      headers: {
        Authorization: `Bearer ${token}`,
        'Content-Type': 'application/json',
        'X-E2E-Run-Id': this.runId,
      },
      body: body ? JSON.stringify(body) : undefined,
    });
  }

  async get<T = unknown>(path: string): Promise<T> {
    const res = await this.send('GET', path);
    if (!res.ok) throw new Error(`GET ${path} ${res.status}: ${await res.text()}`);
    return res.json() as Promise<T>;
  }

  async post<T = unknown>(path: string, body?: unknown): Promise<T> {
    const res = await this.send('POST', path, body);
    if (!res.ok) throw new Error(`POST ${path} ${res.status}: ${await res.text()}`);
    return res.json() as Promise<T>;
  }

  async delete<T = unknown>(path: string): Promise<T> {
    const res = await this.send('DELETE', path);
    if (!res.ok) throw new Error(`DELETE ${path} ${res.status}: ${await res.text()}`);
    return res.json() as Promise<T>;
  }
}
```

```typescript
  // 1. Stripe
  await cancelSubsAndDeleteCustomer(stripeKey, user.email);

  // 2. Backend — DELETE /debug/user-data (idempotent)
  try {
    await user.api.delete('/debug/user-data');
  } catch (err) {
    console.error('[e2e] backend cleanup non-fatal err:', err);
  }

  // 3. Clerk
  if (user.orgId) {
    await deleteOrg({ secretKey: clerkKey, orgId: user.orgId });
  }
  await deleteUser({ secretKey: clerkKey, userId: user.clerkUserId });

  // 4. Settle window before verification (DDB writes take a beat)
  await new Promise((r) => setTimeout(r, 5000));

  // 5. Verification — ALL must pass or we fail the run
  const stripeRemaining = await findCustomerByEmail(stripeKey, user.email);
  if (stripeRemaining) {
    throw new Error(`Stripe leak: customer ${stripeRemaining.id} not deleted`);
  }

  const clerkRemaining = await findUserByEmail({ secretKey: clerkKey, email: user.email });
  if (clerkRemaining) {
    throw new Error(`Clerk leak: user ${clerkRemaining.id} not deleted`);
  }

  // DDB + EFS verification require the API token, but the user is gone now.
  // Solution: backend's /debug/user-data already returned a summary of what
  // it deleted — we trust that summary as long as it returned 200.
  // The Stripe + Clerk checks above are the ones that need post-deletion
  // re-query because those are external systems.
}

export const test = base.extend<{}, { e2eUser: E2EUser }>({
  e2eUser: [
    async ({ browser }, use) => {
      // Constructed once per worker; tests in the same describe share it.
      // Role is set by spec via test.use({ ... }) hack OR per-spec wrapper helpers;
      // simplest is: each spec creates its own fixture-extending file.
      throw new Error('Use createE2EUser directly in beforeAll, not the bare fixture');

    { scope: 'worker' },
```
